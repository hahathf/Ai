def get_neighbors(z):
        return adjac_lis[z]

def heuristic(n):
        return H[n]

def a_star_algorithm(start, stop):
    open_listy = set([start])
    closed_listy = set([])
    # g(n) values
    G = {}
    G[start] = 0
    # parents
    par = {}
    par[start] = start
    while len(open_listy) > 0:
        n = None
        # find node with lowest f(n)
        for z in open_listy:
            if n == None or G[z] + heuristic(z) < G[n] + heuristic(n):
                n = z
        if n == None:
            print('GIVEN PATH DOES NOT EXIST !!!')
            return None
        # if destination reached
        if n == stop:
            path = []
            while par[n] != n:
                path.append(n)
                n = par[n]
            path.append(start)
            path.reverse()
            print('GIVEN PATH IS FOUND : {}'.format(path))
            return path
        # check neighbors
        for (m, dist) in get_neighbors(n):
            if m not in open_listy and m not in closed_listy:
                open_listy.add(m)
                par[m] = n
                G[m] = G[n] + dist
            else:
                if G[m] > G[n] + dist:
                    G[m] = G[n] + dist
                    par[m] = n
                    if m in closed_listy:
                        closed_listy.remove(m)
                        open_listy.add(m)
        print(G)
        open_listy.remove(n)
        closed_listy.add(n)
    print('PATH DOES NOT EXIST !!!')
    return None

start=input("START CITY : ")
goal=input("GOAL CITY : ")
a_star_algorithm(start, goal)
